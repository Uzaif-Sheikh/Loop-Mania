package unsw.loopmania;


import javafx.beans.property.SimpleIntegerProperty;
import javafx.scene.image.Image;
import java.io.File;

public class Armour extends BasicItems implements DefenceStrategy {
    
    private static final double armourPrice = 7.0;

    public Armour(SimpleIntegerProperty x, SimpleIntegerProperty y) {
        super(x, y, armourPrice);
        setDamage("Slug", 0.5);
        setDamage("Zombie", 0.5);
        setDamage("Vampire", 0.5);
        setType("Armour");
        //this.basicImages = new Image((new File("src/images/armour.png")).toURI().toString());
    }
    
    @Override
    public double defendAgainstAttack(Enemy enemy) {
        return getDamage(enemy.getType());
    }
}
