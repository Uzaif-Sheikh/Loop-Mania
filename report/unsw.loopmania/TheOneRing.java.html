package unsw.loopmania;

import javafx.beans.property.SimpleIntegerProperty;
import javafx.scene.image.Image;
import java.io.File;

//Rare items are a part of the world conifguration file.
public class TheOneRing extends RareItems {

    public TheOneRing(SimpleIntegerProperty x, SimpleIntegerProperty y) {
        super(x, y);
        setType("TheOneRing");
        this.rareImage = new Image((new File("src/images/the_one_ring.png")).toURI().toString());
    }

    public void rareItemsHelp(Character curCharacter) {
        curCharacter.refillHealth();
    }
    
}
