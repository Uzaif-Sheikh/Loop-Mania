package unsw.loopmania;
import java.util.HashMap;

import javafx.beans.property.SimpleIntegerProperty;
import javafx.scene.image.Image;

public abstract class BasicItems extends StaticEntity {

    private double price;
    public HashMap<String, Double> itemDamage = new HashMap<String, Double>();
    private String type;
    public Image basicImages;

    public BasicItems(SimpleIntegerProperty x, SimpleIntegerProperty y, double price){
        super(x, y);
        this.price = price;
    }

    public void setDamage(String key,Double value){
        itemDamage.put(key, value);
    }

    public Double getDamage(String key){
        return itemDamage.get(key);
    } 
    
    public double getPrice() {
        return price;
    }

    public void setPrice(double price) {
        this.price = price;
    }

    public String getType() {
        return type;
    }

    public Image getImage(){
        return basicImages;
    }

    public void setType(String type) {
        this.type = type;
    }

}
